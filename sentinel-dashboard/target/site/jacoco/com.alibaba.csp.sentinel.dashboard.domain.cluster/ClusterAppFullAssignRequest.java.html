    private Set<String> remainingList;

        return clusterMap;
    }

    public ClusterAppFullAssignRequest setClusterMap(

        this.clusterMap = clusterMap;
        return this;

        return "ClusterAppFullAssignRequest{" +
            "clusterMap=" + clusterMap +
            ", remainingList=" + remainingList +
            '}';
    }
}
